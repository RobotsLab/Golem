###############################################################################
#
# Golem Plugin/Sensor/FTDAQ library
#
###############################################################################

option(GOLEM_BUILD_PLUGIN_SENSOR_FTDAQ "Build Golem Plugin/Sensor FTDAQ plugin" NO)
mark_as_advanced(GOLEM_BUILD_PLUGIN_SENSOR_FTDAQ)

if(GOLEM_BUILD_PLUGIN_SENSOR AND GOLEM_BUILD_PLUGIN_SENSOR_FTDAQ)
	SET(PACKAGE_DIR "${PACKAGE_PLUGIN_SENSOR}/FTDAQ")

	FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")
	FILE(GLOB_RECURSE PACKAGE_FILES "${PACKAGE_DIR}/resources/*.xml")
	FILE(GLOB_RECURSE PACKAGE_FILES_CAL "${PACKAGE_DIR}/resources/*.cal")

	ADD_LIBRARY(GolemFTDAQ SHARED ${PACKAGE_SOURCES} ${PACKAGE_HEADERS} ${PACKAGE_FILES} ${PACKAGE_FILES_CAL})
	TARGET_COMPILE_OPTIONS(GolemFTDAQ PUBLIC ${CMAKE_DLL_EXPORT_FLAGS})

	TARGET_LINK_LIBRARIES(GolemFTDAQ GolemTools GolemDAQFT)
	
	TARGET_INCLUDE_DIRECTORIES(GolemFTDAQ PUBLIC ${PACKAGE_DIR}/include)

	COPY_FILES(GolemFTDAQ ${RUNTIME_OUTPUT_DIRECTORY} ${PACKAGE_FILES})

	if (CPACK_BUILD_CORE)
		if (WIN32)
			INSTALL(TARGETS GolemFTDAQ RUNTIME DESTINATION bin COMPONENT sensor_execs)
		elseif (UNIX)
			INSTALL(TARGETS GolemFTDAQ LIBRARY DESTINATION bin COMPONENT sensor_execs)
		endif()
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/Sensor/FTDAQ/ COMPONENT sensor_headers)
		INSTALL(FILES ${PACKAGE_FILES} DESTINATION bin COMPONENT sensor_configs)
	endif (CPACK_BUILD_CORE)

	SET_PROPERTY(TARGET GolemFTDAQ PROPERTY PROJECT_LABEL "FTDAQ")
	SET_PROPERTY(TARGET GolemFTDAQ PROPERTY FOLDER "Plugin/Sensor")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
	SOURCE_GROUP("Resource Files" FILES ${PACKAGE_FILES} ${PACKAGE_FILES_CAL})
endif(GOLEM_BUILD_PLUGIN_SENSOR AND GOLEM_BUILD_PLUGIN_SENSOR_FTDAQ)
