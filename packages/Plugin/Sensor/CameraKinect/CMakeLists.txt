###############################################################################
#
# Golem Plugin/Sensor/CameraKinect library
#
###############################################################################

option(GOLEM_BUILD_PLUGIN_SENSOR_CAMERAKINECT "Build Golem Camera Kinect library" NO)
mark_as_advanced(GOLEM_BUILD_PLUGIN_SENSOR_CAMERAKINECT)

if(GOLEM_BUILD_PLUGIN_SENSOR AND GOLEM_BUILD_PLUGIN_SENSOR_CAMERAKINECT)
	SET(PACKAGE_DIR "${PACKAGE_PLUGIN_SENSOR}/CameraKinect")

	# OpenCV
	FIND_PACKAGE(OpenCV REQUIRED)
	# KinectSDK2
	SET(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${GOLEM_PROJECT_CMAKE_MODULES}")
	FIND_PACKAGE(KinectSDK2 REQUIRED)

	FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")
	FILE(GLOB_RECURSE PACKAGE_FILES "${PACKAGE_DIR}/resources/*.xml")
	FILE(GLOB_RECURSE PACKAGE_FILES_CAL "${PACKAGE_DIR}/resources/*.cal")

	ADD_LIBRARY(GolemCameraKinect SHARED ${PACKAGE_SOURCES} ${PACKAGE_HEADERS} ${PACKAGE_FILES} ${PACKAGE_FILES_CAL})		
	TARGET_COMPILE_OPTIONS(GolemCameraKinect PUBLIC ${CMAKE_DLL_EXPORT_FLAGS})

	TARGET_LINK_LIBRARIES(GolemCameraKinect GolemTools ${KINECTSDK2_LIBRARIES})
	
	TARGET_INCLUDE_DIRECTORIES(GolemCameraKinect PUBLIC ${PACKAGE_DIR}/include ${KINECTSDK2_INCLUDE_DIR} ${OPENCV_INCLUDE_DIRS})

	COPY_FILES(GolemCameraKinect ${RUNTIME_OUTPUT_DIRECTORY} ${PACKAGE_FILES})

	if (CPACK_BUILD_CORE)
		if (WIN32)
			INSTALL(TARGETS GolemCameraKinect RUNTIME DESTINATION bin COMPONENT sensor_execs)
		elseif (UNIX)
			INSTALL(TARGETS GolemCameraKinect LIBRARY DESTINATION bin COMPONENT sensor_execs)
		endif()
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/Sensor/CameraKinect/ COMPONENT sensor_headers)
		INSTALL(FILES ${PACKAGE_FILES} DESTINATION bin COMPONENT sensor_configs)
	endif (CPACK_BUILD_CORE)

	SET_PROPERTY(TARGET GolemCameraKinect PROPERTY PROJECT_LABEL "CameraKinect")
	SET_PROPERTY(TARGET GolemCameraKinect PROPERTY FOLDER "Plugin/Sensor")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
	SOURCE_GROUP("Resource Files" FILES ${PACKAGE_FILES} ${PACKAGE_FILES_CAL})
endif(GOLEM_BUILD_PLUGIN_SENSOR AND GOLEM_BUILD_PLUGIN_SENSOR_CAMERAKINECT)
