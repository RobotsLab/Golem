###############################################################################
#
# Golem Plugin/Sensor/CameraPointGrey library
#
###############################################################################

option(GOLEM_BUILD_PLUGIN_SENSOR_CAMERAPOINTGREY "Build Golem Plugin/Sensor Point Grey plugin" NO)
mark_as_advanced(GOLEM_BUILD_PLUGIN_SENSOR_CAMERAPOINTGREY)

if(GOLEM_BUILD_PLUGIN_SENSOR AND GOLEM_BUILD_PLUGIN_SENSOR_CAMERAPOINTGREY)
	SET(PACKAGE_DIR "${PACKAGE_PLUGIN_SENSOR}/CameraPointGrey")

	# FlyCapture2
	SET(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${GOLEM_PROJECT_CMAKE_MODULES}")
	FIND_PACKAGE(FlyCapture2 REQUIRED)
	# OpenCV
	FIND_PACKAGE(OpenCV REQUIRED)

	FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")
	FILE(GLOB_RECURSE PACKAGE_FILES "${PACKAGE_DIR}/resources/*.xml")
	FILE(GLOB_RECURSE PACKAGE_FILES_CAL "${PACKAGE_DIR}/resources/*.cal")

	ADD_LIBRARY(GolemCameraPointGrey SHARED ${PACKAGE_SOURCES} ${PACKAGE_HEADERS} ${PACKAGE_FILES} ${PACKAGE_FILES_CAL})
	TARGET_COMPILE_OPTIONS(GolemCameraPointGrey PUBLIC ${CMAKE_DLL_EXPORT_FLAGS})

	TARGET_LINK_LIBRARIES(GolemCameraPointGrey GolemTools ${FLYCAPTURE2_LIBRARIES})
	
	TARGET_INCLUDE_DIRECTORIES(GolemCameraPointGrey PUBLIC ${PACKAGE_DIR}/include ${FLYCAPTURE2_INCLUDE_DIR} ${OPENCV_INCLUDE_DIRS})

	COPY_FILES(GolemCameraPointGrey ${RUNTIME_OUTPUT_DIRECTORY} ${PACKAGE_FILES})

	if (CPACK_BUILD_CORE)
		if (WIN32)
			INSTALL(TARGETS GolemCameraPointGrey RUNTIME DESTINATION bin COMPONENT sensor_execs)
		elseif (UNIX)
			INSTALL(TARGETS GolemCameraPointGrey LIBRARY DESTINATION bin COMPONENT sensor_execs)
		endif()
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/Sensor/CameraPointGrey/ COMPONENT sensor_headers)
		INSTALL(FILES ${PACKAGE_FILES} DESTINATION bin COMPONENT sensor_configs)
	endif (CPACK_BUILD_CORE)

	SET_PROPERTY(TARGET GolemCameraPointGrey PROPERTY PROJECT_LABEL "CameraPointGrey")
	SET_PROPERTY(TARGET GolemCameraPointGrey PROPERTY FOLDER "Plugin/Sensor")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
	SOURCE_GROUP("Resource Files" FILES ${PACKAGE_FILES} ${PACKAGE_FILES_CAL})
endif(GOLEM_BUILD_PLUGIN_SENSOR AND GOLEM_BUILD_PLUGIN_SENSOR_CAMERAPOINTGREY)
