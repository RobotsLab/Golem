###############################################################################
#
# Golem Plugin/ActiveCtrl/WorkspaceCtrl library
#
###############################################################################

option(GOLEM_BUILD_PLUGIN_ACTIVECTRL_WORKSPACECTRL_SFML "Build Plugin/ActiveCtrl/WorkspaceCtrl SFML joystick support" NO)
mark_as_advanced(GOLEM_BUILD_PLUGIN_ACTIVECTRL_WORKSPACECTRL_SFML)
option(GOLEM_BUILD_PLUGIN_ACTIVECTRL_WORKSPACECTRL_VIRTUOSE6D "Build Plugin/ActiveCtrl/WorkspaceCtrl Virtuose6D haption device support" NO)
mark_as_advanced(GOLEM_BUILD_PLUGIN_ACTIVECTRL_WORKSPACECTRL_VIRTUOSE6D)

if(GOLEM_BUILD_PLUGIN_ACTIVECTRL)
	SET(PACKAGE_DIR "${PACKAGE_PLUGIN_ACTIVECTRL}/WorkspaceCtrl")

	FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")
	FILE(GLOB_RECURSE PACKAGE_FILES "${PACKAGE_DIR}/resources/*.xml")

	ADD_LIBRARY(GolemWorkspaceCtrl SHARED ${PACKAGE_SOURCES} ${PACKAGE_HEADERS} ${PACKAGE_FILES})
	
	TARGET_COMPILE_OPTIONS(GolemWorkspaceCtrl PUBLIC ${CMAKE_DLL_EXPORT_FLAGS})
	if(WIN32)
		if(GOLEM_BUILD_PLUGIN_ACTIVECTRL_WORKSPACECTRL_SFML)
			TARGET_COMPILE_OPTIONS(GolemWorkspaceCtrl PUBLIC /D_GOLEM_BUILD_PLUGIN_ACTIVECTRL_WORKSPACECTRL_SFML)
		endif(GOLEM_BUILD_PLUGIN_ACTIVECTRL_WORKSPACECTRL_SFML)
		if(GOLEM_BUILD_PLUGIN_ACTIVECTRL_WORKSPACECTRL_VIRTUOSE6D)
			TARGET_COMPILE_OPTIONS(GolemWorkspaceCtrl PUBLIC /D_GOLEM_BUILD_PLUGIN_ACTIVECTRL_WORKSPACECTRL_VIRTUOSE6D)
		endif(GOLEM_BUILD_PLUGIN_ACTIVECTRL_WORKSPACECTRL_VIRTUOSE6D)
	elseif(UNIX)
	endif()

	TARGET_LINK_LIBRARIES(GolemWorkspaceCtrl GolemActiveCtrl GolemApp GolemContact)
	if(GOLEM_BUILD_PLUGIN_ACTIVECTRL_WORKSPACECTRL_VIRTUOSE6D)
		TARGET_LINK_LIBRARIES(GolemWorkspaceCtrl GolemCtrlVirtuose6D)
	endif(GOLEM_BUILD_PLUGIN_ACTIVECTRL_WORKSPACECTRL_VIRTUOSE6D)

    TARGET_INCLUDE_DIRECTORIES(GolemWorkspaceCtrl PUBLIC ${PACKAGE_DIR}/include)
	
	COPY_FILES(GolemWorkspaceCtrl ${RUNTIME_OUTPUT_DIRECTORY} ${PACKAGE_FILES})

	if (CPACK_BUILD_CORE)
		if (WIN32)
			INSTALL(TARGETS GolemWorkspaceCtrl RUNTIME DESTINATION bin COMPONENT activectrl_execs)
		elseif (UNIX)
			INSTALL(TARGETS GolemWorkspaceCtrl LIBRARY DESTINATION bin COMPONENT activectrl_execs)
		endif()
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/ActiveCtrl/WorkspaceCtrl/ COMPONENT activectrl_headers)
		INSTALL(FILES ${PACKAGE_FILES} DESTINATION bin COMPONENT activectrl_configs)
	endif (CPACK_BUILD_CORE)

	SET_PROPERTY(TARGET GolemWorkspaceCtrl PROPERTY PROJECT_LABEL "WorkspaceCtrl")
	SET_PROPERTY(TARGET GolemWorkspaceCtrl PROPERTY FOLDER "Plugin/ActiveCtrl")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
	SOURCE_GROUP("Resource Files" FILES ${PACKAGE_FILES})
endif(GOLEM_BUILD_PLUGIN_ACTIVECTRL)
