###############################################################################
#
# Golem Interop/Interop/PCL library
#
###############################################################################

option(GOLEM_BUILD_INTEROP_PCL "Build Golem Interop PCL library" YES)
mark_as_advanced(GOLEM_BUILD_INTEROP_PCL)

if(GOLEM_BUILD_INTEROP AND GOLEM_BUILD_INTEROP_PCL)
	SET(PACKAGE_DIR "${PACKAGE_INTEROP}/Interop/PCL")

	# PCL
	FIND_PACKAGE(PCL REQUIRED)
	list(REMOVE_ITEM PCL_LIBRARIES "vtkproj4") # PCL bug

	FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")

	if(GOLEM_DYNAMIC_LIBS)
		ADD_LIBRARY(GolemInteropPCL SHARED ${PACKAGE_SOURCES} ${PACKAGE_HEADERS})
		if (WIN32)		
			TARGET_COMPILE_OPTIONS(GolemInteropPCL PUBLIC ${CMAKE_DLL_EXPORT_FLAGS})
		elseif (UNIX)
		endif()
	else(GOLEM_DYNAMIC_LIBS)
		ADD_LIBRARY(GolemInteropPCL STATIC ${PACKAGE_SOURCES} ${PACKAGE_HEADERS})
	endif(GOLEM_DYNAMIC_LIBS)

	TARGET_LINK_LIBRARIES(GolemInteropPCL GolemInteropCore ${PCL_LIBRARIES})
	
	TARGET_COMPILE_DEFINITIONS(GolemInteropPCL PUBLIC ${PCL_DEFINITIONS})
	TARGET_INCLUDE_DIRECTORIES(GolemInteropPCL PUBLIC ${PACKAGE_DIR}/include PUBLIC ${GOLEM_PROJECT_PACKAGES}/Interop/Interop/PCL/include/Golem/Interop PUBLIC ${PCL_INCLUDE_DIRS})
	
	if (CPACK_BUILD_CORE)
		if(GOLEM_DYNAMIC_LIBS)
			if (WIN32)
				INSTALL(TARGETS GolemInteropPCL RUNTIME DESTINATION bin COMPONENT core_libs)
			elseif (UNIX)
				INSTALL(TARGETS GolemInteropPCL LIBRARY DESTINATION bin COMPONENT core_libs)
			endif()
		else(GOLEM_DYNAMIC_LIBS)
			INSTALL(TARGETS GolemInteropPCL ARCHIVE DESTINATION lib COMPONENT core_libs)
		endif(GOLEM_DYNAMIC_LIBS)
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/Interop/ COMPONENT core_headers)
	endif (CPACK_BUILD_CORE)
	
	SET_PROPERTY(TARGET GolemInteropPCL PROPERTY PROJECT_LABEL "PCL")
	SET_PROPERTY(TARGET GolemInteropPCL PROPERTY FOLDER "Interop/Interop")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
endif(GOLEM_BUILD_INTEROP AND GOLEM_BUILD_INTEROP_PCL)
