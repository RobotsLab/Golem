###############################################################################
#
# Golem Tools/DAQFT library
#
###############################################################################

option(GOLEM_BUILD_TOOLS_DAQFT "Build Golem Tools DAQFT library" NO)
mark_as_advanced(GOLEM_BUILD_TOOLS_DAQFT)

if(GOLEM_BUILD_TOOLS_DAQFT)
	SET(PACKAGE_DIR "${PACKAGE_TOOLS}/DAQFT")

	# Eigen
	SET(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${GOLEM_PROJECT_CMAKE_MODULES}")
	FIND_PACKAGE(NIDAQmx REQUIRED)

	#FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	SET(PACKAGE_SOURCES
		${PACKAGE_DIR}/src/ATIDAQHardwareInterface.cpp
		${PACKAGE_DIR}/src/DAQFT.cpp
		${PACKAGE_DIR}/src/dom.cpp
		#${PACKAGE_DIR}/src/events.c
		${PACKAGE_DIR}/src/expatls.cpp
		${PACKAGE_DIR}/src/ftconfig.cpp
		${PACKAGE_DIR}/src/ftrt.cpp
		${PACKAGE_DIR}/src/FTSensor.cpp
		${PACKAGE_DIR}/src/node.cpp
		${PACKAGE_DIR}/src/stack.cpp
		${PACKAGE_DIR}/src/Stdafx.cpp
		${PACKAGE_DIR}/src/xmlparse.c
		${PACKAGE_DIR}/src/xmlrole.c
		${PACKAGE_DIR}/src/xmltok.c
		#${PACKAGE_DIR}/src/xmltok_impl.c
		#${PACKAGE_DIR}/src/xmltok_ns.c
	)
	#FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")
	SET(PACKAGE_HEADERS
		${PACKAGE_DIR}/include/Golem/DAQFT/FTSensor.h
	)

	if(GOLEM_DYNAMIC_LIBS)
		ADD_LIBRARY(GolemDAQFT SHARED ${PACKAGE_SOURCES} ${PACKAGE_HEADERS})
		TARGET_COMPILE_OPTIONS(GolemDAQFT PUBLIC ${CMAKE_DLL_EXPORT_FLAGS})
	else(GOLEM_DYNAMIC_LIBS)
		ADD_LIBRARY(GolemDAQFT STATIC ${PACKAGE_SOURCES} ${PACKAGE_HEADERS})
	endif(GOLEM_DYNAMIC_LIBS)

	# Windows 32/64 bit
	if (WIN32)
		TARGET_COMPILE_OPTIONS(GolemDAQFT PUBLIC /D_CRT_SECURE_NO_WARNINGS)
		TARGET_LINK_LIBRARIES(GolemDAQFT ${NIDAQMX_LIBRARIES})
	# Unix Linux/Apple
	elseif (UNIX)
		TARGET_LINK_LIBRARIES(GolemDAQFT ${NIDAQMX_LIBRARIES} rt)
	endif ()
	SET_SOURCE_FILES_PROPERTIES(${PACKAGE_SOURCES} PROPERTIES LANGUAGE CXX)

	TARGET_INCLUDE_DIRECTORIES(GolemDAQFT PUBLIC ${PACKAGE_DIR}/include ${NIDAQMX_INCLUDE_DIR})
		
	if (CPACK_BUILD_CORE)
		if(GOLEM_DYNAMIC_LIBS)
			if (WIN32)
				INSTALL(TARGETS GolemDAQFT RUNTIME DESTINATION bin COMPONENT core_libs)
			elseif (UNIX)
				INSTALL(TARGETS GolemDAQFT LIBRARY DESTINATION bin COMPONENT core_libs)
			endif()
		else(GOLEM_DYNAMIC_LIBS)
			INSTALL(TARGETS GolemDAQFT ARCHIVE DESTINATION lib COMPONENT core_libs)
		endif(GOLEM_DYNAMIC_LIBS)
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/DAQFT/ COMPONENT core_headers)
	endif (CPACK_BUILD_CORE)
	
	SET_PROPERTY(TARGET GolemDAQFT PROPERTY PROJECT_LABEL "DAQFT")
	SET_PROPERTY(TARGET GolemDAQFT PROPERTY FOLDER "Tools")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
endif(GOLEM_BUILD_TOOLS_DAQFT)
