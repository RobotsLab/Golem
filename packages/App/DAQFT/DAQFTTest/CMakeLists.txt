###############################################################################

# Golem App/DAQFT/DAQFTTest application
#
###############################################################################

if(GOLEM_BUILD_APP_DAQFT)
	SET(PACKAGE_DIR "${PACKAGE_APP_DAQFT}/DAQFTTest")

	FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")
	FILE(GLOB_RECURSE PACKAGE_FILES "${PACKAGE_DIR}/resources/*.xml")

	ADD_EXECUTABLE(GolemDAQFTTest ${PACKAGE_SOURCES} ${PACKAGE_HEADERS} ${PACKAGE_FILES})
	TARGET_LINK_LIBRARIES(GolemDAQFTTest GolemDAQFT)
	TARGET_INCLUDE_DIRECTORIES(GolemDAQFTTest PUBLIC ${PACKAGE_DIR}/include)

	COPY_FILES(GolemDAQFTTest ${RUNTIME_OUTPUT_DIRECTORY} ${PACKAGE_FILES})
	SET_PROPERTY(TARGET GolemDAQFTTest PROPERTY RELEASE_POSTFIX ${CMAKE_RELEASE_POSTFIX})
	SET_PROPERTY(TARGET GolemDAQFTTest PROPERTY DEBUG_POSTFIX ${CMAKE_DEBUG_POSTFIX})
	
	if (CPACK_BUILD_CORE)
		INSTALL(TARGETS GolemDAQFTTest RUNTIME DESTINATION bin COMPONENT app_execs)
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/App/DAQFTTest/ COMPONENT app_headers)
		INSTALL(FILES ${PACKAGE_FILES} DESTINATION bin COMPONENT app_configs)
	endif (CPACK_BUILD_CORE)

	SET_PROPERTY(TARGET GolemDAQFTTest PROPERTY PROJECT_LABEL "DAQFTTest")
	SET_PROPERTY(TARGET GolemDAQFTTest PROPERTY FOLDER "App/DAQFT")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
	SOURCE_GROUP("Resource Files" FILES ${PACKAGE_FILES})
endif(GOLEM_BUILD_APP_DAQFT)
