###############################################################################
#
# Golem App/UI/ExPlacement application
#
###############################################################################

option(GOLEM_BUILD_APP_UI_EXPLACEMENT "Build Golem App/UI ExPlacement" NO)
mark_as_advanced(GOLEM_BUILD_APP_UI_EXPLACEMENT)

if(GOLEM_BUILD_APP_UI AND GOLEM_BUILD_APP_UI_EXPLACEMENT)
	SET(PACKAGE_DIR "${PACKAGE_APP_UI}/ExPlacement")

	# OpenCV
	FIND_PACKAGE(OpenCV REQUIRED)

	FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")
	FILE(GLOB_RECURSE PACKAGE_FILES "${PACKAGE_DIR}/resources/*.xml")

	ADD_EXECUTABLE(GolemAppExPlacement ${PACKAGE_SOURCES} ${PACKAGE_HEADERS} ${PACKAGE_FILES})
	TARGET_LINK_LIBRARIES(GolemAppExPlacement GolemApp GolemContact GolemDataImage)
	TARGET_INCLUDE_DIRECTORIES(GolemAppExPlacement PUBLIC ${PACKAGE_DIR}/include ${OPENCV_INCLUDE_DIRS})

	COPY_FILES(GolemAppExPlacement ${RUNTIME_OUTPUT_DIRECTORY} ${PACKAGE_FILES})
	SET_PROPERTY(TARGET GolemAppExPlacement PROPERTY RELEASE_POSTFIX ${CMAKE_RELEASE_POSTFIX})
	SET_PROPERTY(TARGET GolemAppExPlacement PROPERTY DEBUG_POSTFIX ${CMAKE_DEBUG_POSTFIX})
	
	if (CPACK_BUILD_CORE)
		INSTALL(TARGETS GolemAppExPlacement RUNTIME DESTINATION bin COMPONENT app_execs)
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/App/ExPlacement/ COMPONENT app_headers)
		INSTALL(FILES ${PACKAGE_FILES} DESTINATION bin COMPONENT app_configs)
	endif (CPACK_BUILD_CORE)

	SET_PROPERTY(TARGET GolemAppExPlacement PROPERTY PROJECT_LABEL "ExPlacement")
	SET_PROPERTY(TARGET GolemAppExPlacement PROPERTY FOLDER "App/UI")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
	SOURCE_GROUP("Resource Files" FILES ${PACKAGE_FILES})
endif(GOLEM_BUILD_APP_UI AND GOLEM_BUILD_APP_UI_EXPLACEMENT)
