###############################################################################
#
# Golem App/UI/NNLearning application
#
###############################################################################

option(GOLEM_BUILD_APP_UI_NNLEARNING "Build Golem App/UI/NNLearning application" NO)
mark_as_advanced(GOLEM_BUILD_APP_UI_NNLEARNING)

if(GOLEM_BUILD_APP_UI AND GOLEM_BUILD_APP_UI_NNLEARNING)
	SET(PACKAGE_DIR "${PACKAGE_APP_UI}/NNLearning")

	# OpenCV
	FIND_PACKAGE(OpenCV REQUIRED)

	FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")
	FILE(GLOB_RECURSE PACKAGE_FILES "${PACKAGE_DIR}/resources/*.xml")

	ADD_EXECUTABLE(GolemAppNNLearning ${PACKAGE_SOURCES} ${PACKAGE_HEADERS} ${PACKAGE_FILES})
	TARGET_LINK_LIBRARIES(GolemAppNNLearning GolemContact GolemDataImage GolemApp)
	TARGET_INCLUDE_DIRECTORIES(GolemAppNNLearning PUBLIC ${PACKAGE_DIR}/include ${OPENCV_INCLUDE_DIRS})

	COPY_FILES(GolemAppNNLearning ${RUNTIME_OUTPUT_DIRECTORY} ${PACKAGE_FILES})
	SET_PROPERTY(TARGET GolemAppNNLearning PROPERTY RELEASE_POSTFIX ${CMAKE_RELEASE_POSTFIX})
	SET_PROPERTY(TARGET GolemAppNNLearning PROPERTY DEBUG_POSTFIX ${CMAKE_DEBUG_POSTFIX})
	
	if (CPACK_BUILD_CORE)
		INSTALL(TARGETS GolemAppNNLearning RUNTIME DESTINATION bin COMPONENT app_execs)
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/App/NNLearning/ COMPONENT app_headers)
		INSTALL(FILES ${PACKAGE_FILES} DESTINATION bin COMPONENT app_configs)
	endif (CPACK_BUILD_CORE)

	SET_PROPERTY(TARGET GolemAppNNLearning PROPERTY PROJECT_LABEL "NNLearning")
	SET_PROPERTY(TARGET GolemAppNNLearning PROPERTY FOLDER "App/UI")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
	SOURCE_GROUP("Resource Files" FILES ${PACKAGE_FILES})
endif(GOLEM_BUILD_APP_UI AND GOLEM_BUILD_APP_UI_NNLEARNING)
