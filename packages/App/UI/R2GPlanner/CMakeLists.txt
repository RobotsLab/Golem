###############################################################################
#
# Golem App/UI/R2GPlanner application
#
###############################################################################

if(GOLEM_BUILD_APP_UI AND GOLEM_BUILD_PLUGIN_HBPLANNER)
	SET(PACKAGE_DIR "${PACKAGE_APP_UI}/R2GPlanner")

	# OpenCV
	FIND_PACKAGE(OpenCV REQUIRED)

	FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")
	FILE(GLOB_RECURSE PACKAGE_FILES "${PACKAGE_DIR}/resources/*.xml")

	ADD_EXECUTABLE(GolemAppR2GPlanner ${PACKAGE_SOURCES} ${PACKAGE_HEADERS} ${PACKAGE_FILES})
	TARGET_LINK_LIBRARIES(GolemAppR2GPlanner GolemApp GolemHBPlanner GolemContact GolemDataImage GolemDataFeature3D GolemActiveTouchCtrl)
	TARGET_INCLUDE_DIRECTORIES(GolemAppR2GPlanner PUBLIC ${PACKAGE_DIR}/include ${OPENCV_INCLUDE_DIRS})

	COPY_FILES(GolemAppR2GPlanner ${RUNTIME_OUTPUT_DIRECTORY} ${PACKAGE_FILES})
	SET_PROPERTY(TARGET GolemAppR2GPlanner PROPERTY RELEASE_POSTFIX ${CMAKE_RELEASE_POSTFIX})
	SET_PROPERTY(TARGET GolemAppR2GPlanner PROPERTY DEBUG_POSTFIX ${CMAKE_DEBUG_POSTFIX})
	
	if (CPACK_BUILD_CORE)
		INSTALL(TARGETS GolemAppR2GPlanner RUNTIME DESTINATION bin COMPONENT app_execs)
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/App/R2GPlanner/ COMPONENT app_headers)
		INSTALL(FILES ${PACKAGE_FILES} DESTINATION bin COMPONENT app_configs)
	endif (CPACK_BUILD_CORE)

	SET_PROPERTY(TARGET GolemAppR2GPlanner PROPERTY PROJECT_LABEL "R2GPlanner")
	SET_PROPERTY(TARGET GolemAppR2GPlanner PROPERTY FOLDER "App/UI")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
	SOURCE_GROUP("Resource Files" FILES ${PACKAGE_FILES})
endif(GOLEM_BUILD_APP_UI AND GOLEM_BUILD_PLUGIN_HBPLANNER)
