###############################################################################
#
# Golem App/SM/FTServerTest application
#
###############################################################################

option(GOLEM_BUILD_APP_SM_FTSERVERTEST "Build SM FTServer test applications" NO)
mark_as_advanced(GOLEM_BUILD_APP_SM_FTSERVERTEST)

if(GOLEM_BUILD_APP_SM AND GOLEM_BUILD_APP_SM_FTSERVERTEST)
	SET(PACKAGE_DIR "${PACKAGE_APP_SM}/FTServerTest")

	FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")
	FILE(GLOB_RECURSE PACKAGE_FILES "${PACKAGE_DIR}/resources/*.xml")

	ADD_EXECUTABLE(GolemFTServerTest ${PACKAGE_SOURCES} ${PACKAGE_HEADERS} ${PACKAGE_FILES})
	TARGET_LINK_LIBRARIES(GolemFTServerTest GolemSys GolemSM)
	TARGET_INCLUDE_DIRECTORIES(GolemFTServerTest PUBLIC ${PACKAGE_DIR}/include)

	COPY_FILES(GolemFTServerTest ${RUNTIME_OUTPUT_DIRECTORY} ${PACKAGE_FILES})
	SET_PROPERTY(TARGET GolemFTServerTest PROPERTY RELEASE_POSTFIX ${CMAKE_RELEASE_POSTFIX})
	SET_PROPERTY(TARGET GolemFTServerTest PROPERTY DEBUG_POSTFIX ${CMAKE_DEBUG_POSTFIX})
	
	if (CPACK_BUILD_CORE)
		INSTALL(TARGETS GolemFTServerTest RUNTIME DESTINATION bin COMPONENT app_execs)
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/App/FTServerTest/ COMPONENT app_headers)
		INSTALL(FILES ${PACKAGE_FILES} DESTINATION bin COMPONENT app_configs)
	endif (CPACK_BUILD_CORE)

	SET_PROPERTY(TARGET GolemFTServerTest PROPERTY PROJECT_LABEL "FTServerTest")
	SET_PROPERTY(TARGET GolemFTServerTest PROPERTY FOLDER "App/SM")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
	SOURCE_GROUP("Resource Files" FILES ${PACKAGE_FILES})
endif(GOLEM_BUILD_APP_SM AND GOLEM_BUILD_APP_SM_FTSERVERTEST)
