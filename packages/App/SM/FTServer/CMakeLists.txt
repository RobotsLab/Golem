###############################################################################
#
# Golem App/SM/FTServer application
#
###############################################################################

option(GOLEM_BUILD_APP_SM_FTSERVER "Build SM FTServer applications" NO)
mark_as_advanced(GOLEM_BUILD_APP_SM_FTSERVER)

if(GOLEM_BUILD_APP_SM AND GOLEM_BUILD_APP_SM_FTSERVER)
	SET(PACKAGE_DIR "${PACKAGE_APP_SM}/FTServer")

	FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")
	FILE(GLOB_RECURSE PACKAGE_FILES "${PACKAGE_DIR}/resources/*.xml")

	ADD_EXECUTABLE(GolemFTServer ${PACKAGE_SOURCES} ${PACKAGE_HEADERS} ${PACKAGE_FILES})
	TARGET_LINK_LIBRARIES(GolemFTServer GolemSys GolemMath GolemSM GolemDAQFT)
	TARGET_INCLUDE_DIRECTORIES(GolemFTServer PUBLIC ${PACKAGE_DIR}/include)

	COPY_FILES(GolemFTServer ${RUNTIME_OUTPUT_DIRECTORY} ${PACKAGE_FILES})
	SET_PROPERTY(TARGET GolemFTServer PROPERTY RELEASE_POSTFIX ${CMAKE_RELEASE_POSTFIX})
	SET_PROPERTY(TARGET GolemFTServer PROPERTY DEBUG_POSTFIX ${CMAKE_DEBUG_POSTFIX})
	
	if (CPACK_BUILD_CORE)
		INSTALL(TARGETS GolemFTServer RUNTIME DESTINATION bin COMPONENT app_execs)
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/App/FTServer/ COMPONENT app_headers)
		INSTALL(FILES ${PACKAGE_FILES} DESTINATION bin COMPONENT app_configs)
	endif (CPACK_BUILD_CORE)

	SET_PROPERTY(TARGET GolemFTServer PROPERTY PROJECT_LABEL "FTServer")
	SET_PROPERTY(TARGET GolemFTServer PROPERTY FOLDER "App/SM")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
	SOURCE_GROUP("Resource Files" FILES ${PACKAGE_FILES})
endif(GOLEM_BUILD_APP_SM AND GOLEM_BUILD_APP_SM_FTSERVER)
