###############################################################################
#
# Golem App/SM/FTServerSim application
#
###############################################################################

option(GOLEM_BUILD_APP_SM_FTSERVERSIM "Build SM FTServer simulator" NO)
mark_as_advanced(GOLEM_BUILD_APP_SM_FTSERVERSIM)

if(GOLEM_BUILD_APP_SM AND GOLEM_BUILD_APP_SM_FTSERVERSIM)
	SET(PACKAGE_DIR "${PACKAGE_APP_SM}/FTServerSim")

	FILE(GLOB_RECURSE PACKAGE_SOURCES "${PACKAGE_DIR}/src/*.cpp")
	FILE(GLOB_RECURSE PACKAGE_HEADERS "${PACKAGE_DIR}/include/*.h")
	FILE(GLOB_RECURSE PACKAGE_FILES "${PACKAGE_DIR}/resources/*.xml")

	ADD_EXECUTABLE(GolemFTServerSim ${PACKAGE_SOURCES} ${PACKAGE_HEADERS} ${PACKAGE_FILES})
	if (WIN32)
		TARGET_LINK_LIBRARIES(GolemFTServerSim GolemSys GolemSM)
	elseif (UNIX)
		TARGET_LINK_LIBRARIES(GolemFTServerSim GolemSys GolemSM ncurses)
	endif()
	TARGET_INCLUDE_DIRECTORIES(GolemFTServerSim PUBLIC ${PACKAGE_DIR}/include)

	COPY_FILES(GolemFTServerSim ${RUNTIME_OUTPUT_DIRECTORY} ${PACKAGE_FILES})
	SET_PROPERTY(TARGET GolemFTServerSim PROPERTY RELEASE_POSTFIX ${CMAKE_RELEASE_POSTFIX})
	SET_PROPERTY(TARGET GolemFTServerSim PROPERTY DEBUG_POSTFIX ${CMAKE_DEBUG_POSTFIX})
	
	if (CPACK_BUILD_CORE)
		INSTALL(TARGETS GolemFTServerSim RUNTIME DESTINATION bin COMPONENT app_execs)
		INSTALL(FILES ${PACKAGE_HEADERS} DESTINATION include/Golem/App/FTServerSim/ COMPONENT app_headers)
		INSTALL(FILES ${PACKAGE_FILES} DESTINATION bin COMPONENT app_configs)
	endif (CPACK_BUILD_CORE)

	SET_PROPERTY(TARGET GolemFTServerSim PROPERTY PROJECT_LABEL "FTServerSim")
	SET_PROPERTY(TARGET GolemFTServerSim PROPERTY FOLDER "App/SM")
	SOURCE_GROUP("Include Files" FILES ${PACKAGE_HEADERS})
	SOURCE_GROUP("Resource Files" FILES ${PACKAGE_FILES})
endif(GOLEM_BUILD_APP_SM AND GOLEM_BUILD_APP_SM_FTSERVERSIM)
